            const city = document.getElementById('city-filter').value;
            const skill = document.getElementById('skill-filter').value;

        window.saveGig = async () => {
            if(!auth.currentUser) return showPage('auth');

                skill: document.getElementById('gig-skill').value,
                exp: document.getElementById('gig-exp').value,

                ratings: [],
                avgRating: 0,
                userId: auth.currentUser.uid,
                isVerified: false,
                isPinned: false,
                createdAt: Date.now()

                await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'workers'), data);

            // Sort: Pinned First
            const sorted = [...workers].sort((a, b) => (b.isPinned ? 1 : 0) - (a.isPinned ? 1 : 0));

            sorted.forEach(w => {
                const isOwner = auth.currentUser && auth.currentUser.uid === w.userId;
                list.innerHTML += `
                    <div class="bg-white rounded-[3rem] shadow-sm hover:shadow-2xl transition-all border border-slate-100 overflow-hidden relative group">
                        ${w.isPinned ? '<div class="absolute top-6 left-6 z-10"><span class="pinned-badge"><i class="fas fa-thumbtack mr-1"></i> Featured</span></div>' : ''}
                        <div class="p-10">
                            <div class="flex justify-between items-start mb-8">
                                <div>
                                    <h4 class="text-2xl font-black text-slate-800 flex items-center">
                                        ${w.name}
                                        ${w.isVerified ? '<i class="fas fa-check-circle blue-tick"></i>' : ''}
                                    </h4>
                                    <span class="bg-emerald-50 text-emerald-700 px-4 py-1.5 rounded-2xl text-[10px] font-black uppercase tracking-[0.15em] mt-2 inline-block">${w.skill}</span>
                                </div>
                                <div class="cursor-pointer text-center bg-slate-900 text-white px-5 py-3 rounded-[1.5rem] shadow-2xl" onclick="openReviewsModal('${w.id}')">
                                    <p class="font-black text-xl leading-none">★ ${w.avgRating.toFixed(1)}</p>
                                    <p class="text-[8px] font-black uppercase opacity-60 mt-1">${w.ratings?.length || 0} Reviews</p>
                                </div>
                            </div>
                            
                            <div class="space-y-4 mb-10 text-slate-500 font-bold text-sm">
                                <p class="flex items-center"><i class="fas fa-award w-8 text-emerald-500 text-lg"></i> ${w.exp} Sal Experience</p>
                                <p class="flex items-center"><i class="fas fa-map-marker-alt w-8 text-rose-500 text-lg"></i> ${w.city}</p>
                            </div>

                            <div class="flex items-center justify-between pt-8 border-t border-slate-50">
                                <div>
                                    <p class="text-[9px] uppercase text-slate-400 font-black tracking-widest mb-1">Standard Visit</p>

                                        <i class="fab fa-whatsapp text-2xl"></i> <span class="hidden sm:inline">Raabta</span>
                                    </button>
                                </div>
                            </div>

        // --- HIRING & RATINGS ---
        window.handleHire = (id, phone) => {
            if(!auth.currentUser) return showPage('auth');
            currentWorkerId = id;
            const msg = `Asalam o Alaikum! Mein ne Kamm Kar app par aapki profile dekhi hai. Mujhe aapki services chahiye.`;

        window.openRatingModal = (id) => {
            currentWorkerId = id;
            selectedRating = 0;
            document.querySelectorAll('#rating-stars i').forEach(s => s.classList.replace('text-yellow-400', 'text-slate-100'));
            document.getElementById('rating-comment').value = '';
            document.getElementById('rating-modal').classList.remove('hidden');
        };

        window.setRating = (num) => {
            selectedRating = num;
            document.querySelectorAll('#rating-stars i').forEach((s, i) => {
                if(i < num) s.classList.replace('text-slate-100', 'text-yellow-400');
                else s.classList.replace('text-yellow-400', 'text-slate-100');
            });
        };

        window.submitRating = async () => {
            if(!selectedRating) return alert("Select Stars!");
            const comment = document.getElementById('rating-comment').value;
            try {
                const worker = allWorkers.find(w => w.id === currentWorkerId);
                const newRating = { userId: auth.currentUser.uid, userName: auth.currentUser.displayName || "Anonymous", stars: selectedRating, comment, date: Date.now() };
                const updatedRatings = [...(worker.ratings || []), newRating];
                const newAvg = updatedRatings.reduce((a, b) => a + b.stars, 0) / updatedRatings.length;
                await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'workers', currentWorkerId), {
                    ratings: arrayUnion(newRating),
                    avgRating: newAvg
                });
                closeRatingModal();
                alert("Rating Saved!");
            } catch(e) { console.error(e); }
        };

        window.closeRatingModal = () => document.getElementById('rating-modal').classList.add('hidden');

        window.openReviewsModal = (id) => {
            const worker = allWorkers.find(w => w.id === id);
            const list = document.getElementById('reviews-list');
            list.innerHTML = worker.ratings?.length ? '' : '<p class="text-center text-slate-300 py-16 font-black uppercase tracking-widest text-xs">No Reviews Yet</p>';
            worker.ratings?.forEach(r => {
                list.innerHTML += `
                    <div class="bg-slate-50 p-6 rounded-[2rem] border border-slate-100">
                        <div class="flex justify-between items-center mb-4">
                            <p class="font-black text-slate-800 text-sm tracking-tight">${r.userName}</p>
                            <span class="text-yellow-500 font-black">★ ${r.stars}</span>
                        </div>
                        <p class="text-slate-500 text-sm italic font-medium leading-relaxed">"${r.comment || 'Behtareen service!'}"</p>
                    </div>
                `;
            });
            document.getElementById('reviews-modal').classList.remove('hidden');
        };

        window.closeReviewsModal = () => document.getElementById('reviews-modal').classList.add('hidden');

            // Realtime Firebase Listener
            onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'workers'), (snap) => {
                allWorkers = snap.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                renderWorkers(allWorkers);
                renderAdminTable(allWorkers);
                document.getElementById('loading-overlay').classList.add('hidden');
            }, (error) => console.error(error));
        });
